## Diff

Diff:

```diff
- import { Hello } from './hello';
+ import { GoodBye } from './hello';

(function(win) {
-  win['hello'] = new Hello();
+  win['goodbye'] = new GoodBye();
})(this);
```

JS:

```js
- import { Hello } from './hello';
+ import { GoodBye } from './hello';

(function(win) {
-  win['hello'] = new Hello();
+  win['goodbye'] = new GoodBye();
})(this);
```
